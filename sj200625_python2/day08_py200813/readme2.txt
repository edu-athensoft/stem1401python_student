important string functions

count()     - int
find()      - index or -1
rfind()     - index or -1
index()     - index or Error
join()      - string
split()     - list

isalnum()       - True, False
isalpha()       - True, False
isdecimal()     - True, False, exclude \u00B2, \u00BD (superscript, subscript, fractions)
isdigit()       - True, False, include (superscript, subscript)
isnumberic()    - True, False, include (superscript, subscript, fractions)

isidentifier()  - True, False
islower()       - True, False
isupper()       - True, False
isprintable()   - True, False
isspace()       - True, False
istitle()       - True, False
